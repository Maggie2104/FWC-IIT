\item If $\sin y=x\sin(a+y)$, then prove that $\frac{dy}{dx}=\frac{\sin^2(a+y)}{\sin a}$

\item If $\sec^{-1}(\frac{1+x}{1-y})=a$ , then $\frac{dy}{dx}$ is equal to
\begin{enumerate}
\item $\frac{x-1}{y-1}$
\item $\frac{x-1}{y+1}$
\item $\frac{y-1}{x+1}$
\item $\frac{y+1}{x-1}$
\end{enumerate}
